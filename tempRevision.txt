NOTES APP using MONGODB

APP.JS
const express = require('express');
const app = express();
const noteModel = require('./src/models/notes.models');
app.use(express.json());

//create notes
app.post("/create-note", async (req, res) => {
    const data = req.body;

    await noteModel.create({
        title : data.title,
        description : data.description
    })

    res.status(201).json({
        message: "notes posted successfully"
    })
})

//view-notes
app.get("/view-notes", async (req, res) => {
    const notes = await noteModel.find();

    res.status(200).json({
        message : "notes sent successfully",
        notes
    })
})


//update notes
app.patch("/update-notes/:id", async (req, res) => {

    const id = req.params.id;
    const title = req.body.title;
    const description = req.body.description;

    await noteModel.findOneAndUpdate({_id : id}, {title : title, description: description});

    res.status(200).json({
        message : "notes updated successfully",
    })
})

//delete notes
app.delete("/delete-notes/:id", async (req, res) => {

    const id = req.params.id;

    await noteModel.findOneAndDelete({
        _id: id
    })

    res.status(200).json({
        message : "notes successfully deleted"
    })
})


module.exports = app;
